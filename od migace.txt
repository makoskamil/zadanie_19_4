function comments(state = [], action) {
  switch(action.type) {
    case ADD_COMMENT:
      return [
        {
          id: action.id,
          text: action.text,
          votes: 0
        }, 
        ...state.comments
      ];
    case REMOVE_COMMENT:
      return state.filter(comment => comment.id !== action.id);
    case EDIT_COMMENT:
      return state.map(comment => {
        if (comment.id === action.id) {
          comment.text = action.text;
        }
        
        return comment;
      });
    case THUMB_UP:
    case THUMB_DOWN:
   default:
       return state;
}
